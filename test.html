Find Nth Highest salary using Java 8 Stream API

            
                public class EmployeeTest {
        
                    public static void main(String[] args) {
                        List employees = new ArrayList();
                        
                        employees.add(new Employee(100, "Swapnil k", 100000));
                        employees.add(new Employee(200, "MS Dhoni", 90000));
                        employees.add(new Employee(300, "Virat k", 80000));
                        employees.add(new Employee(400, "Sachin T", 90000));
                        employees.add(new Employee(500, "Rohit S", 70000));
                        employees.add(new Employee(600, "Chris G", 1100000));
                        
                        employees.forEach(System.out::println);	
                        
                        //List sortedList
                        
                        Optional salary= employees.stream().map(e->e.getSalary()).sorted(Comparator.reverseOrder()).skip(2).findFirst();//.collect(Collectors.toList());
                        
                        //sortedList.forEach(System.out::println);
                        System.out.println(salary.get());
                    
                    }
                }
